{{- /* Get all pages from the blog section sorted by date (newest first). Returns a slice of dicts: { page, featuredPath } */ -}} {{- $blogPages := where site.RegularPages "Section" "blog" -}} {{- $sortedPages := sort $blogPages "Date" "desc" -}} {{- $result := slice -}} {{- range $sortedPages -}} {{- $featuredPath := partial "tk/path-image-featured" (dict "page" .) -}} {{- $result = $result | append (dict "page" . "featuredPath" $featuredPath) -}} {{- end -}} {{- return $result -}}
